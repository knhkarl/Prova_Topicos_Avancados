CREATE TABLE Auditing (
    IDENTIFIER SERIAL NOT NULL,
    REGISTER_CODE NUMERIC(20) NOT NULL,
    DATETIME TIMESTAMP NOT NULL,
    CONSTRAINT auditing_pk PRIMARY KEY (IDENTIFIER),
    FOREIGN KEY (REGISTER_CODE) REFERENCES Delivery (ORDERCODE)
);

 

CREATE TABLE Delivery (
    ORDERCODE NUMERIC(15) NOT NULL,
    CPF VARCHAR(11) NOT NULL,
    DELIVERY_DATE TIMESTAMP NOT NULL,
    CEP VARCHAR(8) NOT NULL,
    DELIVERY_STATUS VARCHAR(12) NOT NULL,
    CONSTRAINT delivery_pk PRIMARY KEY (ORDERCODE)
);
